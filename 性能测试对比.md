# 对比1
```json

{set:"$output","=":{code:0}},
{set:"start","=":"now()"},

  {set:"$context.data","+=":"index"}

{set:"$output.data","=":"now() - start"},
```

# 对比7
```json
{set:"$context.data","=":[]},
{set:"$output","=":{code:0,data:{}}},
{set:"start","=":"now()"},
{set:"data","=":"1"},
{for:"index",from:"1",to:"1000000",step:"1",loop:[
  {set:"data","=":"index + data"},
  {set:"data","=":"index - data"},
  {set:"data","=":"index * 0.99"},
  {set:"data","=":"data / 0.98"}
]},
{set:"$output.data.data","=":"data"},
{set:"$output.data.time","=":"now() - start"},
{return:"$output"}
```

```ecmascript 6
let data = 1;
for(let i = 1; i <= 1000000; i++){
  data = i + data;
  data = i - data;
  data = i * 0.99;
  data = data / 0.98;
}
```
